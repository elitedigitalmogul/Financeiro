let db = JSON.parse(localStorage.getItem("financas_pro")) || {}; let chart; const $ = id => document.getElementById(id); $("mesInput").value = new Date().toISOString().slice(0,7); function getMes(){ return $("mesInput").value || new Date().toISOString().slice(0,7); } function salvar(){ localStorage.setItem("financas_pro", JSON.stringify(db)); } /* ========================= ADICIONAR ========================= */ function adicionar(){ const mes = getMes(); const nome = $("nome").value; const valor = parseFloat($("valor").value); const categoria = $("categoria").value; if(!nome || isNaN(valor)) return; if(!db[mes]) db[mes] = []; db[mes].push({ id: Date.now(), nome, valor, categoria }); salvar(); $("nome").value=""; $("valor").value=""; atualizarGeral(); } /* ========================= REMOVER ========================= */ function remover(id){ const mes = getMes(); if(!db[mes]) return; db[mes] = db[mes].filter(t => t.id !== id); salvar(); atualizarGeral(); } /* ========================= INSIGHTS (CORRIGIDO) ========================= */ function calcularInsights(transacoes, entradas, saidas){ let texto = ""; let gastos = transacoes.filter(t => t.categoria !== "Renda"); let top = [...gastos] .sort((a,b)=>b.valor-a.valor) .slice(0,3); if(entradas - saidas < 0){ texto += "🚨 Saldo negativo

"; } texto += "🏆 Top 3 gastos:
"; if(top.length === 0){ texto += "- Nenhum gasto
"; } else { top.forEach(t=>{ texto += `- ${t.nome}: R$ ${t.valor.toFixed(2)}
`; }); } return texto; } /* ========================= SCORE (CORRIGIDO) ========================= */ function calcularScore(entradas, saidas, investido){ let score = 50; if(entradas > saidas) score += 20; if(investido > entradas * 0.2) score += 20; if(saidas < entradas * 0.7) score += 10; return Math.min(Math.max(score,0),100); } /* ========================= COMPARAÇÃO MÊS (SEGURA) ========================= */ function compararMes(mes){ let meses = Object.keys(db).sort(); let i = meses.indexOf(mes); if(i <= 0) return "📊 Sem mês anterior"; let atual = db[mes] || []; let anterior = db[meses[i-1]] || []; let gAtual = atual.reduce((a,t)=>a+(t.categoria!=="Renda"?t.valor:0),0); let gAnt = anterior.reduce((a,t)=>a+(t.categoria!=="Renda"?t.valor:0),0); let diff = gAtual - gAnt; if(gAnt === 0) return "📊 Comparação indisponível"; return diff >= 0 ? `📈 +R$ ${diff.toFixed(2)} vs mês anterior` : `📉 -R$ ${Math.abs(diff).toFixed(2)} vs mês anterior`; } /* ========================= MAIN UPDATE (ESTÁVEL) ========================= */ function atualizarGeral(){ const mes = getMes(); const transacoes = db[mes] || []; let entradas = 0; let saidas = 0; let investido = 0; let categorias = { Essencial:0, Estilo:0, Investimento:0 }; let html = ""; transacoes.forEach(t=>{ if(t.categoria === "Renda"){ entradas += t.valor; } else { saidas += t.valor; categorias[t.categoria] = (categorias[t.categoria] || 0) + t.valor; if(t.categoria === "Investimento"){ investido += t.valor; } } html += `
${t.nome} R$ ${t.valor.toFixed(2)}
`; }); $("lista").innerHTML = html; $("kpi-entradas").innerText = "R$ " + entradas.toFixed(2); $("kpi-saidas").innerText = "R$ " + saidas.toFixed(2); $("kpi-saldo").innerText = "R$ " + (entradas - saidas).toFixed(2); /* INSIGHTS */ $("insights").innerHTML = calcularInsights(transacoes, entradas, saidas); /* SCORE */ $("score").innerText = "Score: " + calcularScore(entradas, saidas, investido); /* COMPARAÇÃO */ $("comparacao").innerText = compararMes(mes); /* GRÁFICO */ if(chart) chart.destroy(); chart = new Chart($("chart"),{ type:"doughnut", data:{ labels:["Essencial","Estilo","Investimento"], datasets:[{ data:[ categorias.Essencial, categorias.Estilo, categorias.Investimento ] }] } }); } atualizarGeral();
